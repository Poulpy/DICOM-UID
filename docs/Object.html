def add_missing_dot comp
  raise TypeError unless comp.is_a? String
  if comp[-1] != '.'
    comp << '.'
    return true
  end
  false
end

def odd_byte_rule comp
  return comp[-1].to_i % 2 == 0 || (comp[-2].to_i % 2 == 1 && comp[-1].to_i == 0) if comp.length != 1
  comp[-1].to_i % 2 == 0
end
